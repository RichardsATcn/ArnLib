178 void ArnDiscoverConnector::doClientConnectChanged( int stat, int curPrio)
179 {
180  ArnClient::ConnectStat cs = ArnClient::ConnectStat::fromInt( stat);
181 
182  if (Arn::debugDiscover) qDebug() << "Discover ClientConnectChanged 1: stat=" << stat
183  << " prio=" << curPrio;
184  if (!_resolver || (cs == cs.Connecting)) return;
185  if (cs == cs.Connected) {
186  _resolveRefreshBlocked = false;
187  return;
188  }
189  if (_isResolved || (cs != cs.TriedAll)) { // Resolv ok or still more to try, consider outdated resolv
190  if (curPrio != _discoverHostPrio) return; // Not for resolved host
191  if ((cs != cs.Error) && (cs != cs.Disconnected)) return; // Skip any non error
192  if (Arn::debugDiscover) qDebug() << "Discover ClientConnectChanged 2:";
193 
194  if (_resolveRefreshTime->elapsed() >= _resolveRefreshTimeout * 1000)
195  _resolveRefreshBlocked = false;
196  if (_resolveRefreshBlocked) return;
197  }
198 
199  _resolveRefreshBlocked = true; // Block for further refresh within lockout time
200  _resolveRefreshTime->start();
201 
202  if (Arn::debugDiscover) qDebug() << "Discover ClientConnectChanged 3 resolve: service=" << _arnDisHostService->toString();
203  bool forceUpdate = _isResolved;
204  _resolver->resolve( _arnDisHostService->toString(), forceUpdate); // Do a resolve refresh / retry
205 }

208 void ArnDiscoverConnector::postSetupClient()
209 {
210  if (_resolver)
211  postSetupResolver();
212 
213  QString path;
214  QString connectIdPath = Arn::pathDiscoverConnect + _id + "/";
215 
216  path = connectIdPath + "Status/";
217  ArnItem* arnConnectStatus = new ArnItem( path + "value", this);
218  *arnConnectStatus = _client->connectStatus();
219  connect( _client, SIGNAL(connectionStatusChanged(int,int)), arnConnectStatus, SLOT(setValue(int)));
220  typedef ArnClient::ConnectStat CS;
221  ArnM::setValue( path + "set",
222  QString("%1=Initialized %2=Connecting %3=Connected %4=Connect_error %5=Disconnected")
223  .arg(CS::Init).arg(CS::Connecting).arg(CS::Connected).arg(CS::Error).arg(CS::Disconnected));
224 
225  path = connectIdPath + "Request/";
226  ArnItem* arnConnectReqPV = new ArnItem( path + "value!", this);
227  *arnConnectReqPV = "0";
228  connect( arnConnectReqPV, SIGNAL(changed(int)), this, SLOT(doClientConnectRequest(int)));
229  ArnM::setValue( path + "set", QString("0=Idle 1=Start_connect"));
230 
231  ArnClient::HostList arnHosts = _client->arnList( _directHostPrio);
232  int i = 0;
233  foreach (ArnClient::HostAddrPort host, arnHosts) {
234  path = connectIdPath + "DirectHosts/Host-" + QString::number(i) + "/";
235  ArnItem* hostAddr = new ArnItem( path + "value", _directHosts);
236  ArnItem* hostPort = new ArnItem( path + "Port/value", _directHosts);
237  *hostAddr = host.addr; // Default addr
238  *hostPort = host.port; // Default port
239  hostAddr->addMode( Arn::ObjectMode::Save); // Save mode after default set, will not save default value
240  hostPort->addMode( Arn::ObjectMode::Save);
241  connect( hostAddr, SIGNAL(changed()), this, SLOT(doClientDirHostChanged()));
242  connect( hostPort, SIGNAL(changed()), this, SLOT(doClientDirHostChanged()));
243  ++i;
244  }
245  doClientDirHostChanged(); // Any loaded persistent values will be used
246 
247  connect( _client, SIGNAL(connectionStatusChanged(int,int)), this, SLOT(doClientConnectChanged(int,int)));
248  _hasBeenSetupClient = true;
249 
250  if (!_resolver)
251  doClientReadyToConnect( _client, _id);
252 }

263 void ArnDiscoverConnector::doClientDirHostChanged()
264 {
265  QObjectList dirHostOList = _directHosts->children();
266  int dirHostOListSize = dirHostOList.size();
267  Q_ASSERT((dirHostOListSize & 1) == 0);
268 
270  _client->clearArnList( _directHostPrio);
271  for (int i = 0; i < dirHostOListSize / 2; ++i) {
272  ArnItem* hostAddr = qobject_cast<ArnItem*>( dirHostOList.at( 2 * i + 0));
273  Q_ASSERT(hostAddr);
274  ArnItem* hostPort = qobject_cast<ArnItem*>( dirHostOList.at( 2 * i + 1));
275  Q_ASSERT(hostPort);
276  _client->addToArnList( hostAddr->toString(), quint16( hostPort->toInt()), _directHostPrio);
277  }

290 void ArnDiscoverConnector::postSetupResolver()
291 {
292  QString path;
293  QString connectIdPath = Arn::pathDiscoverConnect + _id + "/";
294 
295  path = connectIdPath + "DiscoverHost/";
296  _arnDisHostServicePv = new ArnItem( path + "Service/value!", _resolver);
297  _arnDisHostService = new ArnItem( path + "Service/value", _resolver);
298  _arnDisHostAddress = new ArnItem( path + "Host/value", _resolver);
299  _arnDisHostPort = new ArnItem( path + "Host/Port/value", _resolver);
300  _arnDisHostStatus = new ArnItem( path + "Status/value", _resolver);
301  typedef ArnZeroConf::Error Err;
302  ArnM::setValue( path + "Status/set",
303  QString("%1=Resolved %2=Resolving %3=Bad_request_sequence %4=Resolv_timeout "
304  "%5=Unicast_DNS_Fail")
305  .arg(Err::Ok).arg(Err::Running).arg(Err::BadReqSeq).arg(Err::Timeout)
306  .arg(Err::UDnsFail));
307 
308  *_arnDisHostServicePv = _resolver->defaultService(); // Use this default if no active persistent service
309  _arnDisHostService->addMode( Arn::ObjectMode::Save); // Save mode after default set, will not save default value
310  if (_service.isEmpty()) // Non empty _service is always used
311  _service = _arnDisHostService->toString(); // Otherwise persistent/deafult value will be used as request
312  *_arnDisHostService = _service;
313 
314  connect( _arnDisHostServicePv, SIGNAL(changed()), this, SLOT(doClientServicetChanged()));
315  connect( _resolver, SIGNAL(infoUpdated(int,ArnDiscoverInfo::State)),
316  this, SLOT(doClientResolvChanged(int,ArnDiscoverInfo::State)));
317 
318  doClientServicetChanged(); // Perform request
319 }

335 void ArnDiscoverConnector::doClientResolvChanged( int index, ArnDiscoverInfo::State state)
336 {
337  Q_ASSERT(_arnDisHostService);
338  Q_ASSERT(_arnDisHostAddress);
339  Q_ASSERT(_arnDisHostPort);
340 
341  const ArnDiscoverInfo& info = _resolver->infoByIndex( index);
342  if (info.serviceName() != _arnDisHostService->toString()) return; // Not the current service
343 
344  *_arnDisHostAddress = info.hostWithInfo();
345  *_arnDisHostPort = info.hostPortString();
346  *_arnDisHostStatus = info.resolvCode();
347 
348  if (state <= state.ServiceName) { // New resolv has started
349  if (Arn::debugDiscover) qDebug() << "ArnDiscoverConnector New resolv started: service=" << info.serviceName();
350  _isResolved = false;
351  _client->clearArnList( _discoverHostPrio);
352  }
353  else if (state == state.HostIp) {
354  _isResolved = true;
355  _client->clearArnList( _discoverHostPrio);
356  _client->addToArnList( info.hostWithInfo(), info.hostPort(), _discoverHostPrio);
357  if (_client->connectStatus() == ArnClient::ConnectStat::Init) {
358  doClientReadyToConnect( _client, _id);
359  }
360  }
361  else if (info.isError()) {
362  _client->clearArnList( _discoverHostPrio);
363  if (_client->connectStatus() == ArnClient::ConnectStat::Init) {
364  doClientReadyToConnect( _client, _id);
365  }
366  }
367 }
