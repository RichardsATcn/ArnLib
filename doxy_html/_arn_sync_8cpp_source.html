112 ArnItemNet* ArnSync::newNetItem( const QString& path, Arn::ObjectSyncMode syncMode, bool* isNewPtr)
113 {
114  ArnItemNet* itemNet = new ArnItemNet( this);
115  if (!itemNet->open( path)) return 0;
116 
117  uint netId = itemNet->linkId(); // Use clients linkId as netID for this Item
118  if (_itemNetMap.contains( netId)) { // Item is already synced by this client
119  if (isNewPtr) { // Allow duplicate ref, indicate this is not new
120  delete itemNet;
121  itemNet = _itemNetMap.value( netId, 0);
122  itemNet->addSyncMode( syncMode, true);
123  *isNewPtr = false;
124  return itemNet;
125  }
126  else { // Not allow duplicate ref, return error;
127  qDebug() << "Arn netSync Item already synced: path=" << itemNet->path();
128  delete itemNet;
129  return 0;
130  }
131  }
132  if (isNewPtr)
133  *isNewPtr = true;
134 
135  itemNet->addSyncMode( syncMode, true);
136  setupItemNet( itemNet, netId);
137  itemNet->setBlockEcho( true); // Client gives no echo to avoid endless looping
138  _syncQueue.enqueue( itemNet);
139  // qDebug() << "Sync EnQueue: id=" << itemNet->netId() << " path=" << itemNet->path();
140 
141  if (!_isSending) sendNext();
142 
143  return itemNet;

147 void ArnSync::socketInput()
148 {
149  _dataReadBuf.resize( _socket->bytesAvailable());
150  int nbytes = _socket->read( _dataReadBuf.data(), _dataReadBuf.size());
151  if (nbytes <= 0) {
152  return; // No bytes / error
153  }
154  _dataReadBuf.resize( nbytes);
155  _dataRemain += _dataReadBuf;
156 
157  QByteArray xString;
158  int pos;
159  while ((pos = _dataRemain.indexOf("\n")) >= 0) {
160  // Set xString to string before \n
161  xString.resize(0);
162  xString.append( _dataRemain.constData(), pos);
163  _dataRemain.remove(0, pos + 1); // Set remain to string after \n
164 
165  xString.replace('\r', ""); // Remove any \r
166  _commandMap.fromXString( xString); // Load command Map
167  _replyMap.clear(); // Reset reply Map
168 
169  if (Arn::debugRecInOut) qDebug() << "Rec-in: " << xString;
170  doCommand();
171 
172  if (_replyMap.size()) {
173  sendXSMap( _replyMap);
174  // _replySendingCount++;
175  // cout << "REPLY: |" << _replyMap.toXString() << "|" << endl;
176  }
177  }

181 void ArnSync::doCommand()
182 {
183  // int stat = 0;
184  uint stat = ArnError::Ok;
185  QByteArray command = _commandMap.value(0);
186 
188  if (command.startsWith('R')) {
189  emit replyRecord( _commandMap);
190  }
192  else if (command == "flux") {
193  stat = doCommandFlux();
194  }
195  else if (command == "event") {
196  stat = doCommandEvent();
197  }
198  else if (command == "get") {
199  stat = doCommandGet();
200  }
201  else if (command == "set") {
202  stat = doCommandSet();
203  }
204  else if (command == "sync") {
205  stat = doCommandSync();
206  }
207  else if (command == "mode") {
208  stat = doCommandMode();
209  }
210  else if (command == "destroy") {
211  stat = doCommandDestroy();
212  }
213  else if (command == "ls") {
214  stat = doCommandLs();
215  }
216  else if (command == "ver") {
217  _replyMap.add(ARNRECNAME, "Rver").add("data", "Arn ver 1.0");
218  }
219  else if (command == "exit") {
220  _socket->disconnectFromHost();
221  }
223  else if (command == "err") {
224  qDebug() << "REC-ERR: |" << _commandMap.toXString() << "|";
225  }
226  else {
227  stat = ArnError::RecUnknown;
228  _replyMap.add(ARNRECNAME, "err");
229  _replyMap.add("data", QByteArray("Unknown record:") + command);
230  }
231  if (_replyMap.size()) {
232  _replyMap.add("stat", QByteArray::number( stat));
233  }
234 }

237 uint ArnSync::doCommandSync()
238 {
239  QByteArray path = _commandMap.value("path");
240  QByteArray smode = _commandMap.value("smode");
241  uint netId = _commandMap.value("id").toUInt();
242 
243  ArnItemNet* itemNet = new ArnItemNet;
244  if (!itemNet->open( path)) {
245  delete itemNet;
246  return ArnError::CreateError;
247  }
248 
249  setupItemNet( itemNet, netId);
250  itemNet->addSyncModeString( smode, false); // SyncMode is only for the item (session), not the link
251 
252  Arn::ObjectSyncMode syncMode = itemNet->syncMode();
253  if (syncMode.is( syncMode.Monitor)) {
254  setupMonitorItem( itemNet);
255  }
256  if (!itemNet->getModeString().isEmpty()) { // If non default mode
257  itemNet->modeUpdate(); // Make server send the current mode to client
258  }
259  if ((itemNet->type() != Arn::DataType::Null)
260  && !(itemNet->syncMode().is( syncMode.Master))) { // Only send non Null Value to non master
261  itemNet->itemUpdate( ArnLinkHandle()); // Make server send the current value to client
262  }
263 
264  return ArnError::Ok;
265 }

420 void ArnSync::connected()
421 {
422  _isConnected = true;
423  _syncQueue.clear();
424 
426  ArnItemNet* itemNet;
427  QByteArray mode;
428  QMapIterator<uint,ArnItemNet*> i( _itemNetMap);
429  while (i.hasNext()) {
430  i.next();
431  itemNet = i.value();
432  _syncQueue.enqueue( itemNet);
433  mode = itemNet->getModeString();
434  // If non default mode that isn't already in modeQueue
435  if (!mode.isEmpty() && !itemNet->isDirtyMode()) {
436  _modeQueue.enqueue( itemNet);
437  }
438  if ((itemNet->type() != Arn::DataType::Null) // Only send non Null Value ...
439  && (!itemNet->isPipeMode()) // from non pipe ..
440  && (itemNet->syncMode().is( Arn::ObjectSyncMode::Master))) { // which is master
441  itemNet->itemUpdate( ArnLinkHandle()); // Make client send the current value to server
442  }
443  }
444  sendNext();
445 }

542 void ArnSync::addToFluxQue( const ArnLinkHandle& handleData)
543 {
544  ArnItemNet* itemNet = qobject_cast<ArnItemNet*>( sender());
545  if (!itemNet) {
546  ArnM::errorLog( QString(tr("Can't get ArnItemNet sender for itemChanged")),
547  ArnError::Undef);
548  return;
549  }
550 
551  if (itemNet->isPipeMode()) {
552  if (itemNet->isOnlyEcho()) {
553  // qDebug() << "Flux skip pipe echo: path=" << itemNet->path() << " data=" << itemNet->arnExport()
554  // << "itemId=" << itemNet->itemId();
555  itemNet->resetDirty(); // Arm for more updates
556  return; // Don't send any echo to a Pipe
557  }
558 
559  FluxRec* fluxRec = getFreeFluxRec();
560  fluxRec->xString += makeFluxString( itemNet, handleData);
561  itemNet->resetDirty();
562  if (handleData.has( ArnLinkHandle::QueueFindRegexp)) {
563  QRegExp rx = handleData.valueRef( ArnLinkHandle::QueueFindRegexp).toRegExp();
564  // qDebug() << "AddFluxQueue Pipe QOW: rx=" << rx.pattern();
565  int i;
566  for (i = 0; i < _fluxPipeQueue.size(); ++i) {
567  FluxRec*& fluxRecQ = _fluxPipeQueue[i];
568  _syncMap.fromXString( fluxRecQ->xString);
569  QString fluxDataStrQ = _syncMap.valueString("data");
570  if (rx.indexIn( fluxDataStrQ) >= 0) { // Match
571  // qDebug() << "AddFluxQueue Pipe QOW match: old:"
572  // << fluxRecQ->xString << " new:" << fluxRec->xString;
573  _fluxRecPool += fluxRecQ; // Free item to be replaced
574  fluxRecQ = fluxRec;
575  i = -1; // Mark match
576  break;
577  }
578  }
579  if (i >= 0) { // No match
580  // qDebug() << "AddFluxQueue Pipe QOW nomatch:"
581  // << fluxRec->xString;
582  _fluxPipeQueue.enqueue( fluxRec);
583  }
584  }
585  else { // Normal Pipe
586  // qDebug() << "AddFluxQueue Pipe:"
587  // << fluxRec->xString;
588  _fluxPipeQueue.enqueue( fluxRec);
589  }
590  }
591  else { // Normal Item
592  if (itemNet->syncMode().is( Arn::ObjectSyncMode::Master)
593  && itemNet->isOnlyEcho())
594  {
595  itemNet->resetDirty(); // Arm for more updates
596  return; // Don't send any echo to a Master
597  }
598 
599  itemNet->setQueueNum( ++_queueNumCount);
600  _fluxItemQueue.enqueue( itemNet);
601  }
602 
603  if (!_isSending) {
604  sendNext();
605  }
606 }

681 void ArnSync::sendNext()
682 {
683  _isSending = false;
684 
685  if ( !_isConnected || !_socket->isValid()) {
686  return;
687  }
688 
689  ArnItemNet* itemNet;
690 
691  if (!_syncQueue.isEmpty()) {
692  itemNet = _syncQueue.dequeue();
693  sendSyncItem( itemNet);
694  _isSending = true;
695  }
696  else if (!_modeQueue.isEmpty()) {
697  itemNet = _modeQueue.dequeue();
698  sendModeItem( itemNet);
699  itemNet->resetDirtyMode();
700  _isSending = true;
701  }
702  else { // Flux queues - send entity with lowest queue number
703  int itemQueueNum = _fluxItemQueue.isEmpty() ? _queueNumDone + INT_MAX : _fluxItemQueue.head()->queueNum();
704  int pipeQueueNum = _fluxPipeQueue.isEmpty() ? _queueNumDone + INT_MAX : _fluxPipeQueue.head()->queueNum;
705  int itemQueueRel = itemQueueNum - _queueNumDone;
706  int pipeQueueRel = pipeQueueNum - _queueNumDone;
707 
708  if ((itemQueueRel < INT_MAX) || (pipeQueueRel < INT_MAX)) { // At least 1 flux queue not empty
709  if (itemQueueRel < pipeQueueRel) { // Item flux queue
710  _queueNumDone = itemQueueNum;
711 
712  itemNet = _fluxItemQueue.dequeue();
713  sendFluxItem( itemNet);
714  itemNet->resetDirty();
715  }
716  else { // Pipe flux queue
717  _queueNumDone = pipeQueueNum;
718 
719  FluxRec* fluxRec = _fluxPipeQueue.dequeue();
720  _fluxRecPool += fluxRec;
721  send( fluxRec->xString);
722  }
723  _isSending = true;
724  }
725  }
