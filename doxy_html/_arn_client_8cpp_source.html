41 ArnClient::ArnClient(QObject *parent) :
42  QObject(parent)
43 {
44  _arnMountPoint = 0;
45  _isAutoConnect = false;
46  _retryTime = 2;
47  _port = 0;
48 
49  _socket = new QTcpSocket( this);
50  _arnNetSync = new ArnSync( _socket, true, this);
51  _connectTimer = new QTimer( this);
52 
53  connect( _socket, SIGNAL(connected()), this, SIGNAL(tcpConnected()));
54  connect( _socket, SIGNAL(disconnected()), this, SIGNAL(tcpDisConnected()));
55  connect( _arnNetSync, SIGNAL(replyRecord(XStringMap&)), this, SLOT(doReplyRecord(XStringMap&)));
56  connect( _arnNetSync, SIGNAL(replyRecord(XStringMap&)), this, SIGNAL(replyRecord(XStringMap&)));
57 
58  connect( _socket, SIGNAL(error(QAbstractSocket::SocketError)),
59  this, SLOT(tcpError(QAbstractSocket::SocketError)));
60  connect( _connectTimer, SIGNAL(timeout()), this, SLOT(reConnectArn()));
61 
62  // Special mount point for sys
63  ArnItem* sysMountPoint = new ArnItem("/.sys/", this);
64  connect( sysMountPoint, SIGNAL(arnItemCreated(QString)), this, SLOT(createNewItem(QString)));
65 }

155 ArnItemNet* ArnClient::newNetItem( QString path, ArnItem::SyncMode syncMode, bool* isNewPtr)
156 {
157  if (ArnM::isMainThread()) {
158  return _arnNetSync->newNetItem( path, syncMode, isNewPtr);
159  }
160  else { // Threaded - must be threadsafe
161  ArnThreadComCaller threadCom;
162 
163  threadCom.p()->_retObj = 0; // Just in case ...
164  if (gDebugThreading) qDebug() << "newNetItem-thread: start path=" << path;
165  QMetaObject::invokeMethod( this,
166  "newNetItemProxy",
167  Qt::QueuedConnection,
168  Q_ARG( ArnThreadCom*, threadCom.p()),
169  Q_ARG( QString, path),
170  Q_ARG( int, syncMode.f),
171  Q_ARG( void*, isNewPtr));
172  threadCom.waitCommandEnd(); // Wait main-thread gives retObj
173  ArnItemNet* retItemNet = qobject_cast<ArnItemNet*>( threadCom.p()->_retObj);
174  if (retItemNet) if (gDebugThreading) qDebug() << "newNetItem-thread: end path=" << retItemNet->path();
175 
176  return retItemNet;
177  }
178 }

208 void ArnClient::doReplyRecord( XStringMap& replyMap)
209 {
210  QByteArray reply = replyMap.value(0);
211 
212  if (reply == "Rget") {
213  emit replyGet( replyMap.valueString("data"), replyMap.valueString("path"));
214  }
215  else if (reply == "Rset") {
216  }
217  else if (reply == "Rsync") {
218  }
219  else if (reply == "Rls") {
220  emit replyLs( makeItemList( replyMap), replyMap.valueString("path"));
221  }
222  else if (reply == "Rver") {
223  emit replyVer( replyMap.valueString("data"));
224  }
225  else {
226  }
227 }
