155 bool ArnClient::addMountPoint( const QString& localPath, const QString& remotePath)
156 {
157  if (Arn::debugShareObj) qDebug() << "Adding mount point: localPath=" << localPath
158  << " remotePath=" << remotePath;
159  if (localPath.isEmpty()) {
160  return false;
161  }
162  QString localPath_ = Arn::fullPath( localPath);
163 
165  foreach (const MountPointSlot& mpSlot, _mountPoints) {
166  QString mplPath = mpSlot.localPath;
167  if (localPath_.startsWith( mplPath) || mplPath.startsWith( localPath_)) {
168  ArnM::errorLog( QString(tr("Mount points not exclusive: new=")) +
169  localPath + " existing=" + mplPath,
170  ArnError::Undef);
171  return false;
172  }
173  }
174 
175  MountPointSlot mpSlot;
176  mpSlot.arnMountPoint = new ArnItem( this);
177  bool isOk = mpSlot.arnMountPoint->openFolder( localPath_);
178  if (isOk) {
179  mpSlot.localPath = localPath_;
180  mpSlot.remotePath = remotePath.isEmpty() ? localPath_ : Arn::fullPath( remotePath);
181  connect( mpSlot.arnMountPoint, SIGNAL(arnItemCreated(QString)), this, SLOT(createNewItem(QString)));
182  _mountPoints += mpSlot;
183  mpSlot.arnMountPoint->setReference( &_mountPoints.last()); // Give a ref to this added slot
184  }
185  else
186  delete mpSlot.arnMountPoint;
187 
188  return isOk;
189 }

192 bool ArnClient::removeMountPoint( const QString& localPath)
193 {
194  QString localPath_ = Arn::fullPath( localPath);
195 
196  int mpSize = _mountPoints.size();
197  for (int i = 0; i < mpSize; ++ i) {
198  const MountPointSlot& mountPoint = _mountPoints.at(i);
199  if (mountPoint.localPath == localPath_) {
200  if (mountPoint.arnMountPoint)
201  delete mountPoint.arnMountPoint;
202  _mountPoints.removeAt(i);
203  return true;
204  }
205  }
206  return false;
207 }

292 ArnItemNet* ArnClient::newNetItem( QString path, Arn::ObjectSyncMode syncMode, bool* isNewPtr)
293 {
294  if (ArnM::isMainThread()) {
295  QString path_ = Arn::fullPath( path);
296  MountPointSlot mpSlot;
297  foreach (const MountPointSlot& mountPoint, _mountPoints) {
298  if (path_.startsWith( mountPoint.localPath)) {
299  mpSlot = mountPoint;
300  break;
301  }
302  }
303  return _arnNetSync->newNetItem( path_, mpSlot.localPath, mpSlot.remotePath, syncMode, isNewPtr);
304  }
305  else { // Threaded - must be threadsafe
306  ArnThreadComCaller threadCom;
307 
308  threadCom.p()->_retObj = 0; // Just in case ...
309  if (Arn::debugThreading) qDebug() << "newNetItem-thread: start path=" << path;
310  QMetaObject::invokeMethod( this,
311  "newNetItemProxy",
312  Qt::QueuedConnection,
313  Q_ARG( ArnThreadCom*, threadCom.p()),
314  Q_ARG( QString, path),
315  Q_ARG( int, syncMode.toInt()),
316  Q_ARG( void*, isNewPtr));
317  threadCom.waitCommandEnd(); // Wait main-thread gives retObj
318  ArnItemNet* retItemNet = qobject_cast<ArnItemNet*>( threadCom.p()->_retObj);
319  if (retItemNet) if (Arn::debugThreading) qDebug() << "newNetItem-thread: end path=" << retItemNet->path();
320 
321  return retItemNet;
322  }
323 }

337 void ArnClient::tcpError(QAbstractSocket::SocketError socketError)
338 {
339  QString errTextSum = QString(tr("TCP Client Msg:")) + _socket->errorString();
340  ArnM::errorLog( errTextSum, ArnError::ConnectionError);
341  emit tcpError( _socket->errorString(), socketError);

344  emit connectionStatusChanged( _connectStat, _curPrio);

371 void ArnClient::doConnectArnLogic()
372 {
373  QString arnHost;
374  quint16 port = 0;
375  int curPrio = -1;
376  _curPrio = curPrio;
377 
378  if (_nextHost < 0) { // Normal single host connect
379  arnHost = _arnHost;
380  port = _port;
381  }
382  else if (!_hostTab.isEmpty()) { // Arn connection list
383  if (_nextHost >= _hostTab.size()) { // Past end of list, restart
384  _nextHost = 0;
385  emit connectionStatusChanged( ConnectStat::TriedAll, -1);
386  }
387 
388  const HostAddrPort& slot = _hostTab.at( _nextHost);
389  arnHost = slot.addr;
390  port = slot.port;
391  curPrio = _hostPrioTab.at( _nextHost);
392  // qDebug() << "ArnClient connectlogic: hostTabSize=" << _hostTab.size() << " index=" << _nextHost
393  // << " prio=" << _hostPrioTab.at( _nextHost);
394  ++_nextHost;
395  }
396 
397  if (arnHost.isEmpty()) return;
398 
399  if (port == 0)
400  port = Arn::defaultTcpPort;
401 
402  _socket->abort();
403  _socket->connectToHost( Arn::hostFromHostWithInfo( arnHost), port);
404  _curConnectAP.addr = arnHost;
405  _curConnectAP.port = port;
406  _curPrio = curPrio;
407  _connectStat = ConnectStat::Connecting;
408 
409  emit connectionStatusChanged( _connectStat, _curPrio);
410 }
