199 ArnItemPersist* ArnPersist::getPersistItem( QString path)
200 {
201  ArnItemPersist* item = new ArnItemPersist( this);
202  item->open( path);
203  uint linkId = item->linkId();
204 
205  if (_itemPersistMap.contains( linkId)) { // Already as persist
206  qDebug() << "getPersistItem already persist: path=" << path << " link=" << linkId;
207  delete item;
208  item = _itemPersistMap.value( linkId);
209 
210  return item;
211  }
212 
213  item->setIgnoreSameValue( false);
214  item->setDelay( 5000);
215 
216  connect( item, SIGNAL(changed()), this, SLOT(doArnUpdate()));
217  connect( item, SIGNAL(arnLinkDestroyed()), this, SLOT(doArnDestroy()));
218  _itemPersistMap.insert( linkId, item);
219 
220  return item;

224 ArnItemPersist* ArnPersist::setupMandatory( QString path, bool isMandatory)
225 {
226  ArnItemPersist::StoreType st;
227  ArnItemPersist* item = 0;
228 
229  if (isMandatory) {
230  item = getPersistItem( path);
231  item->setMandatory(true);
232  if ((item->storeType() == st.DataBase) && !_pathPersistMap.contains( path)) {
233  _pathPersistMap.insert( path, item->linkId());
234  item->addMode( ArnItem::Mode::Save);
235  }
236  }
237  else {
238  uint linkId = _pathPersistMap.value( path);
239  if (!linkId) return 0;
240 
241  item = _itemPersistMap.value( linkId);
242  Q_ASSERT(item);
243  item->setMandatory(false);
244  if (item->storeType() != st.File) { // Not persist file
245  _pathPersistMap.remove( path); // Map only needed for mandatory & file
246  if (!item->getMode().is( ArnItem::Mode::Save)) {
247  _itemPersistMap.remove( linkId);
248  delete item;
249  }
250  }
251  }
252  return item;
253 }

283 void ArnPersist::doArnModeChanged( QString path, uint linkId, ArnItem::Mode mode)
284 {
285  qDebug() << "Persist modeChanged: path=" << path << " mode=" << mode.f;
286  if (!mode.is( mode.Save)) return; // Not save-mode
287  if (_itemPersistMap.contains( linkId)) return; // Already in map
288  qDebug() << "Persist modeChanged New: path=" << path << " mode=" << mode.f;
289 
290  ArnItemPersist* item = getPersistItem( path);
291  QByteArray data;
292  int storeId;
293  if (getDbValue( path, data, storeId)) { // path exist in db, load data
294  qDebug() << "Persist modeChanged dbRead: path=" << path;
295  item->arnImport( data, false);
296  item->setStoreId( storeId);
297  }
298  else { // path is new in db, create it with null data
299  qDebug() << "Persist modeChanged dbInsert: path=" << path;
300  insertDbValue( path, QByteArray());
301  if (getDbId( path, storeId)) {
302  item->setStoreId( storeId);
303  }
304  item->setValue( QByteArray(), false); // Do a null update, to signal update done
305  }
306 }

309 void ArnPersist::doArnUpdate()
310 {
311  ArnItemPersist* item;
312  item = qobject_cast<ArnItemPersist*>( sender());
313  if (!item) {
314  Arn::errorLog( QString(tr("Can't get ArnItemPersist sender for doArnUpdate")),
315  ArnError::Undef);
316  return;
317  }
318 
319  switch (item->storeType()) {
320  case ArnItemPersist::StoreType::DataBase:
321  {
322  int storeId = item->storeId();
323  if (storeId) {
324  updateDbValue( storeId, item->arnExport());
325  }
326  else {
327  Arn::errorLog( QString(tr("Can't get persist storeId for doArnUpdate: path=")) +
328  item->path(), ArnError::Undef);
329  }
330  break;
331  }
332  case ArnItemPersist::StoreType::File:
333  {
334  QString relPath = item->path( ArnLink::NameF::Relative);
335  qDebug() << "Persist arnUpdate: Save to relPath=" << relPath;
336  QFile file( _persistDir->absoluteFilePath( relPath));
337  file.open( QIODevice::WriteOnly);
338  QByteArray data = item->toByteArray();
339  file.write( data);
340  // qDebug() << "Persist arnUpdate: Save to file=" << file.fileName();
341  break;
342  }
343  }
344  // qDebug() << "Persist Update: id=" << storeId << " path=" << item->path() << " value=" << item->toByteArray();
345 }

569 void ArnPersist::doLoadMandatory()
570 {
571  QList<int> mandatoryList;
572  if (!getDbMandatoryList( mandatoryList)) return; // Nothing to load
573 
574  QString path;
575  QByteArray value;
576  foreach (int storeId, mandatoryList) {
577  if (!getDbValue( storeId, path, value)) continue; // Not ok ...
578 
579  ArnItemPersist::StoreType st;
580  ArnItemPersist* item = setupMandatory( path, true);
581  Q_ASSERT(item);
582  if (item->storeType() != st.DataBase) continue; // Not a database persist
583 
584  QByteArray data;
585  int storeId;
586  if (getDbValue( path, data, storeId)) {
587  item->arnImport( data, false);
588  item->setStoreId( storeId);
589  }
590  else {
591  item->setValue( QByteArray(), false); // Do a null update, to signal update done
592  }
593  }
