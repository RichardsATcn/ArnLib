312 void ArnItem::itemUpdated( const ArnLinkHandle& handleData, const QByteArray* value)
313 {
314  if (!value) { // Update of item with no data supplied
315  if (_delayTimer) {
316  if (!_delayTimer->isActive()) {
317  _delayTimer->start();
318  }
319  }
320  else {
321  doItemUpdate( handleData);
322  }
323  }
324  else { // Update of item with data supplied (pipe in multi-thread)
325  if (_emitChanged) {
326  emit changed();
327  }
328  if (_emitChangedInt) {
329  emit changed( int( value->toInt()));
330  }
331  if (_emitChangedDouble) {
332  emit changed( double( value->toDouble()));
333  }
334  if (_emitChangedBool) {
335  emit changed( bool( value->toInt() != 0));
336  }
337  if (_emitChangedString) {
338  emit changed( QString::fromUtf8( value->constData(), value->size()));
339  }
340  if (_emitChangedByteArray) {
341  emit changed( *value);
342  }
343  if (_emitChangedVariant) {
344  // Can only handle printable value ...
345  emit changed( QVariant( QString::fromUtf8( value->constData(), value->size())));
346  }
347  resetOnlyEcho(); // Nothing else yet ...
348  }
349 }

352 void ArnItem::doItemUpdate( const ArnLinkHandle& handleData)
353 {
354  Q_UNUSED(handleData);
355 
356  if (_delayTimer ) {
357  _delayTimer->stop();
358  }
359 
360  if (_emitChanged) {
361  emit changed();
362  }
363  if (_emitChangedInt) {
364  emit changed( toInt());
365  }
366  if (_emitChangedDouble) {
367  emit changed( toDouble());
368  }
369  if (_emitChangedBool) {
370  emit changed( toBool());
371  }
372  if (_emitChangedString) {
373  emit changed( toString());
374  }
375  if (_emitChangedByteArray) {
376  emit changed( toByteArray());
377  }
378  if (_emitChangedByteArray) {
379  emit changed( toVariant());
380  }
381  resetOnlyEcho(); // Nothing else yet ...
382 }
