62 void ArnMonitor::setMonitorPath( QString path, ArnClient *client)
63 {
64  if (client) _arnClient = client;
65  _monitorPath = path;
66  if (!_monitorPath.endsWith("/")) _monitorPath += "/";
67 
68  if (_arnClient) {
69  // No recursion due to no emit of arnItemCreated as this is a folder
70  // ItemNet is not threadsafe, only use in connect etc
71  //_itemNet = _arnClient->newNetItem( _monitorPath, ArnItem::SyncMode::Monitor);

73  _itemNet = _arnClient->newNetItem( _monitorPath, ArnItem::SyncMode::Normal, &isNew);
74  if (!_itemNet) return;

76  connect( _itemNet, SIGNAL(arnEvent(QByteArray,QByteArray,bool)),

119 void ArnMonitor::dispatchArnEvent( QByteArray type, QByteArray data, bool isLocal)
120 {
121  if (isLocal) return; // Don't handle local events
122 
123  QString foundPath;
124  // "created" (new fresh) is special case of "found"
125  if (type == "itemCreated") {
126  foundPath = QString::fromUtf8( data.constData(), data.size());
127  emit arnItemCreated( foundPath);
128  }
129  else if (type == "itemFound") {
130  foundPath = QString::fromUtf8( data.constData(), data.size());
131  }
132 
133  QString childPath = ArnM::childPath( _monitorPath, foundPath);
134  // qDebug() << "### arnMon dispArnEv: childPath=" << childPath;
135  if (!childPath.isEmpty()) {
136  if (!_foundChilds.contains( childPath)) { // Only send events about unknown items
137  _foundChilds += childPath;
138  // qDebug() << "### arnMon dispArnEv: Emit chFound childPath=" << childPath;
139  emit arnChildFound( childPath);
140  if (childPath.endsWith('/')) emit arnChildFoundFolder( childPath);
141  else emit arnChildFoundLeaf( childPath);
142  }
143  }
144 }
