85 {
86  id = QObject::qt_metacall( call, id, arguments);
87  if ((id < 0) || (call != QMetaObject::InvokeMetaMethod))
88  return id;

90  Q_ASSERT(id < _slotTab.size());

102  args[0],
103  args[1],
104  args[2],

230 void ArnRpc::addSenderSignals( QObject* sender, QString prefix)
231 {
232  const QMetaObject* metaObject = sender->metaObject();
233  int methodCount = metaObject->methodCount();
234  QByteArray methodSignCompHead;
235  for (int i = 0; i < methodCount; ++i) {
236  const QMetaMethod& method = metaObject->method(i);
237  if (method.methodType() != QMetaMethod::Signal) continue;
238 
239  QByteArray methodSign = method.signature();
240  if (!methodSign.startsWith( prefix.toAscii())) continue;
241 
242  QByteArray methodName = methodSign.left( methodSign.indexOf('('));
243  QByteArray funcName = methodName.mid( prefix.size());
244 
245  QByteArray methodSignComp = methodSign;
246  methodSignComp.chop(1); // Remove last ")"
247  if (!_mode.is( Mode::NoDefaultArgs) // When using Default args ...
248  && methodSignCompHead.startsWith( methodSignComp)) // Same method with less param
249  continue; // Skip it, otherwise multiple rpc calls.
250 
251  methodSignCompHead = methodSignComp;
252  _dynamicSignals->addSignal( sender, i, funcName);
253  }
254 }

276 bool ArnRpc::invoke( const QString& funcName,
277  MQGenericArgument arg1,
278  MQGenericArgument arg2,
279  MQGenericArgument arg3,
280  MQGenericArgument arg4,
281  MQGenericArgument arg5,
282  MQGenericArgument arg6,
283  MQGenericArgument arg7,
284  MQGenericArgument arg8)
285 {
286  if (!_pipe || !_pipe->isOpen()) {
287  errorLog( QString(tr("Pipe not open")),
288  ArnError::RpcInvokeError);
289  return false;
290  }
291 
292  XStringMap xsmCall;
293  if (_legacy)
294  xsmCall.add("func", funcName.toAscii());
295  else
296  xsmCall.add("", funcName.toAscii());
297 
298  int nArg = 0;
299  bool stat = true; // Default ok
300  stat &= xsmAddArg( xsmCall, arg1, 1, nArg);
301  stat &= xsmAddArg( xsmCall, arg2, 2, nArg);
302  stat &= xsmAddArg( xsmCall, arg3, 3, nArg);
303  stat &= xsmAddArg( xsmCall, arg4, 4, nArg);
304  stat &= xsmAddArg( xsmCall, arg5, 5, nArg);
305  stat &= xsmAddArg( xsmCall, arg6, 6, nArg);
306  stat &= xsmAddArg( xsmCall, arg7, 7, nArg);
307  stat &= xsmAddArg( xsmCall, arg8, 8, nArg);
308  if (_legacy)
309  xsmCall.add("n", QByteArray::number( nArg));
310 
311  if (stat) {
312  *_pipe = xsmCall.toXString();
313  }
314  return stat;
315 }

318 bool ArnRpc::xsmAddArg( XStringMap& xsm, const MQGenericArgument& arg, uint index, int& nArg)
319 {
320  if (!arg.name() || !arg.label() || !arg.data()) return true; // Empty arg
321  if (nArg + 1 != int( index)) return true; // Out of seq - Finished
322 
324  bool isListType = false;
325  QByteArray typeName = arg.name();
326  if (typeName == "QStringList") {
327  isListType = true;
328  }
329  int type = QMetaType::type( typeName.constData());
330  if (!type) {
331  errorLog( QString(tr("Unknown type:") + typeName.constData()),
332  ArnError::RpcInvokeError);
333  return false;
334  }
335 
337  QByteArray argLabel = arg.label();
338  if (!argLabel.isEmpty()) {
339  _legacy = false;
340  }
341 
343  QByteArray argDataDump;
344  QStringList argDataList;
345  bool isBinaryType = false;
346  QVariant varArg( type, arg.data());
347  if (isListType) {
348  argDataList = varArg.toStringList();
349  }
350  else if (varArg.canConvert( QVariant::String)) {
351  argDataDump = varArg.toString().toUtf8();
352  }
353  else {
354  QDataStream stream( &argDataDump, QIODevice::WriteOnly);
355  if (!QMetaType::save( stream, type, arg.data())) {
356  errorLog( QString(tr("Can't export type:") + typeName.constData()),
357  ArnError::RpcInvokeError);
358  return false;
359  }
360  isBinaryType = true;
361  }
362 
363  if (isBinaryType && _legacy)
364  xsm.add("f", index, QByteArray("@"));
365 
367  QByteArray argPrefix;
368  if (_legacy)
369  xsm.add("type", index, typeName);
370  else if (typeName == "int")
371  argPrefix = "int";
372  else if (typeName == "uint")
373  argPrefix = "uint";
374  else if (typeName == "bool")
375  argPrefix = "bool";
376  else if (typeName == "QByteArray")
377  argPrefix = "ba";
378  else if (typeName == "QStringList")
379  argPrefix = "list";
380  else if (typeName != "QString")
381  xsm.add((isBinaryType ? "tb" : "t"), typeName);
382 
384  if (_legacy)
385  xsm.add("arg", index, argDataDump);
386  else {
387  if (argPrefix.isEmpty())
388  argPrefix = "a";
389  if (!argLabel.isEmpty())
390  argPrefix += ".";
391  if (isListType) { // Handle list (QStringList)
392  int i = 0;
393  QString argData;
394  if (!argDataList.isEmpty() && !argDataList.at(0).isEmpty()) {
395  argData = argDataList.at(0);
396  ++i;
397  }
398  xsm.add( argPrefix + argLabel, argData);
399  // Output each element in list
400  for(; i < argDataList.size(); ++i) {
401  argData = argDataList.at(i);
402  if (argData.isEmpty())
403  xsm.add("le", "");
404  else
405  xsm.add("", argData);
406  }
407  }
408  else {
409  xsm.add( argPrefix + argLabel, argDataDump);
410  }
411  }
412 
413  ++nArg;
414  return true; // 1 arg added
415 }

418 void ArnRpc::pipeInput( QByteArray data)
419 {
420  if (data.startsWith('"')) { // Text is received
421  int endSize = data.endsWith('"') ? (data.size() - 1) : data.size();
422  if (endSize < 1)
423  endSize = 1;
424  emit textReceived( QString::fromUtf8( data.mid( 1, endSize - 1), endSize - 1));
425  return;
426  }
427 
428  XStringMap xsmCall( data);
429  QByteArray rpcFunc = xsmCall.value(0);
430  if (rpcFunc == "$help") { // Built in help
431  return funcHelp( xsmCall);
432  }
433 
434  QGenericArgument args[10];
435  int argc = 0;
436 
437  // qDebug() << "rpc pipeInput: data=" << data;
438  QByteArray methodName = _methodPrefix + rpcFunc;
439  _legacy &= !xsmCall.key(0).isEmpty(); // Detect legacy or new
440 
441  if (_isIncludeSender) {
442  args[ argc++] = Q_ARG( ArnRpc*, this);
443  }
444 
445  int argStart = argc;
446  bool stat = true; // Default ok
447  int index = 1; // Start after function name in xsm
448  while (index > 0) {
449  if ((index >= xsmCall.size()) || (xsmCall.key( index) == "n"))
450  break; // End of args
451  if (argc > 9) {
452  errorLog( QString(tr("To many args:") + QString::number( argc))
453  + tr(" method:") + methodName.constData(),
454  ArnError::RpcReceiveError);
455  return;
456  }
457  stat = xsmLoadArg( xsmCall, args[ argc++], index, methodName);
458  if (!stat) break;
459  }
460 
461  if (stat) {
462  _receiverStorage->_rpcSender = this;
463  stat = QMetaObject::invokeMethod( _receiver,
464  methodName.constData(),
465  Qt::AutoConnection,
466  args[0],
467  args[1],
468  args[2],
469  args[3],
470  args[4],
471  args[5],
472  args[6],
473  args[7],
474  args[8],
475  args[9]);
476  _receiverStorage->_rpcSender = 0;
477  if(!stat) {
478  errorLog( QString(tr("Can't invoke method:")) + methodName.constData(),
479  ArnError::RpcReceiveError);
480  sendText("Can't invoke method, use $help");
481  }
482  }
483 
485  for (int i = argStart; i < argc; ++i) {
486  int type = QMetaType::type( args[i].name());
487  if (args[i].data()) {
488  QMetaType::destroy( type, args[i].data());
489  }
490  }
491 }

494 bool ArnRpc::xsmLoadArg( const XStringMap& xsm, QGenericArgument& arg, int &index,
495  const QByteArray& methodName)
496 {
497  static const QByteArray qlistName = "QStringList";
498  static const QByteArray qstringName = "QString";
499  static const QByteArray qbaName = "QByteArray";
500  static const QByteArray intName = "int";
501  static const QByteArray uintName = "uint";
502  static const QByteArray boolName = "bool";
503 
505  const QByteArray* typeName = &qstringName; // Default type
506  bool isBinaryType = false;
507  bool isListType = false;
508  bool argInType = false;
509  if (xsm.keyRef( index).startsWith("f")) { // Legacy
510  isBinaryType = true;
511  ++index;
512  }
513  const QByteArray& typeKey = xsm.keyRef( index);
514  if (typeKey == "tb") {
515  isBinaryType = true;
516  typeName = &xsm.valueRef( index);
517  ++index;
518  }
519  else if (typeKey.startsWith("t")) {
520  typeName = &xsm.valueRef( index);
521  ++index;
522  }
523  else if (typeKey.startsWith("int")) {
524  typeName = &intName;
525  argInType = true;
526  }
527  else if (typeKey.startsWith("uint")) {
528  typeName = &uintName;
529  argInType = true;
530  }
531  else if (typeKey.startsWith("bool")) {
532  typeName = &boolName;
533  argInType = true;
534  }
535  else if (typeKey.startsWith("ba")) {
536  typeName = &qbaName;
537  argInType = true;
538  }
539  else if (typeKey.startsWith("li")) {
540  typeName = &qlistName;
541  isListType = true;
542  argInType = true;
543  }
544  int type = QMetaType::type( typeName->constData());
545  if (!type) {
546  errorLog( QString(tr("Unknown type:") + typeName->constData())
547  + tr(" method:") + methodName.constData(),
548  ArnError::RpcReceiveError);
549  return false;
550  }
551  //qDebug() << "--- typeName=" << typeName->constData() << " type=" << type;
552 
554  forever {
555  if (index >= xsm.size()) {
556  index = -1; // Mark no more args
557  return true;
558  }
559  const QByteArray key = xsm.keyRef( index);
560  if ( argInType || (key == "") || key.startsWith("a"))
561  break; // Found arg
562  ++index;
563  }
564  const QByteArray& argDataDump = xsm.valueRef( index);
565  QStringList argDataList;
566  if (isListType) { // Handle list (QStringList)
567  if (!argDataDump.isEmpty())
568  argDataList += QString::fromUtf8( argDataDump.constData(), argDataDump.size());
569  ++index;
570  while (index < xsm.size()) {
571  const QByteArray key = xsm.keyRef( index);
572  if ((key != "") && (key != "le"))
573  break;
574  const QByteArray arg = xsm.valueRef( index);
575  argDataList += QString::fromUtf8( arg.constData(), arg.size());
576  ++index;
577  }
578  }
579  else {
580  ++index;
581  }
582 
584  if (isListType) {
585  QVariant varArg( argDataList);
586  void* argData = QMetaType::construct( type, varArg.constData());
587  Q_ASSERT( argData);
588  arg = QGenericArgument( typeName->constData(), argData); // Assign arg as it has been allocated
589  }
590  else if (isBinaryType) {
591  void* argData = QMetaType::construct( type);
592  Q_ASSERT( argData);
593  arg = QGenericArgument( typeName->constData(), argData); // Assign arg as it has been allocated
594  QDataStream stream( argDataDump);
595  if (!QMetaType::load( stream, type, argData)) {
596  errorLog( QString(tr("Can't' import ds type:") + typeName->constData())
597  + tr(" method:") + methodName.constData(),
598  ArnError::RpcReceiveError);
599  return false;
600  }
601  }
602  else {
603  QVariant varArg( QString::fromUtf8( argDataDump.constData(), argDataDump.size()));
604  if (!varArg.convert( QVariant::Type( type))) {
605  errorLog( QString(tr("Can't' import str type:") + typeName->constData())
606  + tr(" method:") + methodName.constData(),
607  ArnError::RpcReceiveError);
608  return false;
609  }
610  void* argData = QMetaType::construct( type, varArg.constData());
611  Q_ASSERT( argData);
612  arg = QGenericArgument( typeName->constData(), argData); // Assign arg as it has been allocated
613  }
614 
615  return true;
616 }

619 void ArnRpc::funcHelp( const XStringMap &xsm)
620 {
621  Q_UNUSED(xsm)
622 
623  const QMetaObject* metaObject = _receiver->metaObject();
624  int methodIndexHead = -1;
625  int parCountMin = 10;
626  QByteArray methodNameHead;
627  QByteArray methodSignHead;
628  int methodCount = metaObject->methodCount();
629  for (int i = 0; i < methodCount; ++i) {
630  const QMetaMethod& method = metaObject->method(i);
631  QByteArray methodSign = method.signature();
632  if (!methodSign.startsWith( _methodPrefix)) continue;
633 
634  methodSign.chop(1); // Remove last ")"
635  QList<QByteArray> parNames = method.parameterNames();
636  int parCount = parNames.size();
637 
638  if (methodSignHead.startsWith( methodSign)) { // Same method with less param
639  parCountMin = parCount;
640  }
641  else {
642  if (methodIndexHead >= 0)
643  funcHelpMethod( metaObject->method( methodIndexHead),
644  methodNameHead, parCountMin);
645  methodIndexHead = i;
646  methodSignHead = methodSign;
647  methodNameHead = methodSign.left( methodSign.indexOf('('));
648  parCountMin = parCount;
649  }
650  }
651  if (methodIndexHead >= 0)
652  funcHelpMethod( metaObject->method( methodIndexHead), methodNameHead, parCountMin);
653  sendText("$help");
654 }

657 void ArnRpc::funcHelpMethod( const QMetaMethod &method, QByteArray name, int parNumMin)
658 {
659  QString line = QString::fromAscii( name.mid( _methodPrefix.size()));
660 
661  QList<QByteArray> typesNames = method.parameterTypes();
662  QList<QByteArray> parNames = method.parameterNames();
663 
664  bool wasListType = false;
665  int parCount = parNames.size();
666  for (int i = _isIncludeSender; i < parCount; ++i) {
667  QByteArray param;
668  QByteArray parName = parNames.at(i);
669  QByteArray typeName = typesNames.at(i);
670 
671  int type = QMetaType::type( typeName.constData());
672  QVariant varPar( type);
673  bool isBinaryType = (type == 0) || !varPar.canConvert( QVariant::String);
674  bool isListType = false;
675  QByteArray parType;
676  if (typeName == "int")
677  parType = "int";
678  else if (typeName == "uint")
679  parType = "uint";
680  else if (typeName == "bool")
681  parType = "bool";
682  else if (typeName == "QByteArray")
683  parType = "ba";
684  else if (typeName == "QStringList") {
685  parType = "list";
686  isListType = true;
687  }
688  else if (typeName == "QString")
689  parType = wasListType ? "a" : "";
690  else {
691  parType = "a";
692  param += (isBinaryType ? "tb" : "t") + QByteArray("=") + typeName + " ";
693  }
694 
695  if (!parType.isEmpty())
696  param += parType + "=";
697  param += "<" + parName + ">";
698  if (i >= parNumMin)
699  param = "[" + param + "]";
700  line += " " + QString::fromAscii( param);
701 
702  wasListType = isListType;
703  }
704  sendText( line);
705 }

722 void ArnRpc::errorLog( QString errText, ArnError err, void* reference)
723 {
724  QString idText;

732 void ArnRpc::batchConnect( const QObject *sender, const QRegExp &rgx,
733  const QObject *receiver, const QString& replace,
734  Mode mode)
735 {
736  QList<QByteArray> signalSignTab;
737  QList<QByteArray> methodSignLowTab;
738  if (mode.is( mode.Debug))
739  qDebug() << "batchConnect: regExp=" << rgx.pattern() << " replace=" << replace;
740 
742  const QMetaObject* metaObject = sender->metaObject();
743  QList<QByteArray> doneMethodSignTab;
744  int methodCount = metaObject->methodCount();
745  for (int i = 0; i < methodCount; ++i) {
746  const QMetaMethod& method = metaObject->method(i);
747  if (method.methodType() != QMetaMethod::Signal) continue; // Must be a Signal
748 
749  QByteArray signalSign = method.signature();
750  if (doneMethodSignTab.contains( signalSign)) continue; // Already done (inherited)
751  doneMethodSignTab += signalSign;
752 
753  QString signalName = QString::fromAscii( signalSign.left( signalSign.indexOf('(')).constData());
754  QByteArray paramSign = signalSign.mid( signalName.size());
755  if (rgx.indexIn( signalName) >= 0) { // Match of signal in regExp
756  QString methodName = replace;
757  for (int j = 1; j <= rgx.numCaptures(); ++j) {
758  methodName.replace("\\" + QString::number(j), rgx.cap(j));
759  }
760  signalSignTab += signalSign;
761  QByteArray methodSignLow = (methodName.toAscii() + paramSign).toLower();
762  methodSignLowTab += methodSignLow;
763  if (mode.is( mode.Debug))
764  qDebug() << "batchConnect: try match signal=" << signalSign <<
765  " method(low)=" << methodSignLow;
766  }
767  else if (mode.is( mode.Debug))
768  qDebug() << "batchConnect: No regExp match on signal=" << signalSign;
769  }
770 
772  metaObject = receiver->metaObject();
773  methodCount = metaObject->methodCount();
774  doneMethodSignTab.clear();
775  QByteArray methodSignCompHead;
776  for (int i = 0; i < methodCount; ++i) {
777  const QMetaMethod& method = metaObject->method(i);
778  QByteArray methodSign = method.signature();
779  if (doneMethodSignTab.contains( methodSign)) continue; // Already done (inherited)
780  doneMethodSignTab += methodSign;
781 
782  QByteArray methodSignComp = methodSign;
783  methodSignComp.chop(1); // Remove last ")"
784  if (!mode.is( mode.NoDefaultArgs) // When using Default args ...
785  && methodSignCompHead.startsWith( methodSignComp)) // Same method with less param
786  continue; // Skip it, otherwise multiple calls.
787  methodSignCompHead = methodSignComp;
788 
789  QByteArray methodSignLow = methodSign.toLower();
790  int index = methodSignLowTab.indexOf( methodSignLow);
791  if (index >= 0) { // Match signal to method
792  const char* methodPrefix = (method.methodType() == QMetaMethod::Signal) ? "2" : "1";
793  connect( sender, "2" + signalSignTab[ index],
794  receiver, methodPrefix + methodSign);
795  if (mode.is( mode.Debug))
796  qDebug() << "batchConnect: connect signal=" << signalSignTab[ index] <<
797  " method=" << methodSign;
798  }
799  else if (mode.is( mode.Debug))
800  qDebug() << "batchConnect: No match on method=" << methodSignLow;
801  }
802 }
