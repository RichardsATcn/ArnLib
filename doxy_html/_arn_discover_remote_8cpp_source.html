57 void ArnDiscoverRemote::startUseServer( ArnServer* arnServer, ArnDiscover::Type discoverType)
58 {
59  QHostAddress addr = arnServer->listenAddress();
60  QString listenAddr = ((addr == QHostAddress::Any) || (addr == QHostAddress::AnyIPv6))
61  ? QString("Any") : addr.toString();
62  int hostPort = arnServer->port();
63  ArnM::setValue( Arn::pathDiscoverThis + "Interfaces/Listen/value", listenAddr);
64  ArnM::setValue( Arn::pathDiscoverThis + "Host/value", QHostInfo::localHostName());
65  ArnM::setValue( Arn::pathDiscoverThis + "Host/Port/value", hostPort);
66  ArnM::loadFromDirRoot( Arn::pathDiscover + "help.xhtml", QDir( Arn::resourceArnRoot), Arn::Coding::Text);
67 
69  int i = 0;
70  foreach (QNetworkInterface interface, QNetworkInterface::allInterfaces()) {
71  QNetworkInterface::InterfaceFlags flags = interface.flags();
72  if (!flags.testFlag( QNetworkInterface::IsUp)
73  || flags.testFlag( QNetworkInterface::IsPointToPoint)
74  || flags.testFlag( QNetworkInterface::IsLoopBack))
75  continue;
76 
77  foreach (QNetworkAddressEntry entry, interface.addressEntries()) {
78  QAbstractSocket::NetworkLayerProtocol prot = entry.ip().protocol();
79  if ((prot != QAbstractSocket::IPv4Protocol) && (prot != QAbstractSocket::IPv6Protocol))
80  continue;
81 
82  QString path = (Arn::pathDiscoverThis + "Interfaces/Interf-%1/").arg(i);
83  QString addr = entry.ip().toString();
84  QString name = interface.humanReadableName();
85  ArnM::setValue( path + "addr", addr);
86  ArnM::setValue( path + "name", name);
87  ArnM::setValue( path + "value", addr + " [" + name + "]");
88  ++i;
89  }
90  }
91 
92  // Setup advertise, but don't start yet, can be waiting for service name
93  ArnDiscoverAdvertise::advertiseService( discoverType, service(), hostPort);
94 }

118 void ArnDiscoverRemote::postSetupThis()
119 {
120  _servTimer->start( _initialServiceTimeout * 1000); // If no service name set, give time for external setting ...
121  connect( _servTimer, SIGNAL(timeout()), this, SLOT(serviceTimeout()));
122 
123  connect( &_arnService, SIGNAL(changed(QString)), this, SLOT(firstServiceSetup(QString)));
124  connect( &_arnServicePv, SIGNAL(changed(QString)), this, SLOT(firstServiceSetup(QString)));
125  QString servicePath = Arn::pathDiscoverThis + "Service/value";
126  _arnServicePv.open( Arn::twinPath( servicePath));
127  _arnService.addMode( Arn::ObjectMode::Save);
128  _arnService.open( servicePath);
129  // Any loaded persistent service name has now been sent to firstServiceSetup()
130 
131  ArnDiscoverAdvertise::postSetupThis();
132 }

143 void ArnDiscoverRemote::firstServiceSetup( QString serviceName, bool forceSetup)
144 {
145  QString useService = service(); // Priority for any previous set service name
146  if (useService.isEmpty())
147  useService = serviceName;
148  if (Arn::debugDiscover) qDebug() << "firstServiceSetup: service=" << serviceName
149  << " useService=" << useService << " forceSetup=" << forceSetup;
150 
151  if (!forceSetup && useService.isEmpty()) return; // Try later (at timeout)
152 
153  _servTimer->stop();
154  disconnect( &_arnService, SIGNAL(changed(QString)), this, SLOT(firstServiceSetup(QString)));
155  disconnect( &_arnServicePv, SIGNAL(changed(QString)), this, SLOT(firstServiceSetup(QString)));
156  connect( &_arnServicePv, SIGNAL(changed(QString)), this, SLOT(doServiceChanged(QString)));
157 
158  setService( useService);
159 }

180 void ArnDiscoverRemote::setService( QString service)
181 {
182  if (Arn::debugDiscover) qDebug() << "DiscoverRemote setService: serviceName=" << service;
183 
184  if (hasSetupAdvertise()) {
185  bool isAdvertise = state().isAny( State::Advertise);
186  _arnService.setValue( service, isAdvertise ? Arn::SameValue::Ignore
187  : Arn::SameValue::Accept);
188  // If advertising not yet has started, any value must trigger the "service changed logic"
189  }
