186 void ArnDepend::startMonitor()
187 {
188  if (_started) return; // Already started
189 
190  // Connect & check all depency slots
191  foreach( DepSlot* slot, _depTab) {
192  connect( &slot->arnEchoPipe, SIGNAL(changed(QString)), this, SLOT(echoCheck(QString)));
193  echoCheck("", slot);
194  }
195  _started = true;
196 }

210 void ArnDepend::echoCheck( QString echo, DepSlot* slot)
211 {
212  if (Arn::debugDepend) qDebug() << "echoCheck: monitorName=" << _name;
213  if (slot == 0) {
214  ArnItem* arnItem = qobject_cast<ArnItem*>( sender());
215  if (arnItem) slot = static_cast<DepSlot*>( arnItem->reference());
216  }
217  if (slot == 0) {
218  ArnM::errorLog( QString(tr("Can't get slot for DependEchoCheck monitor=")) + _name,
219  ArnError::Undef);
220  return;
221  }
222  if (slot->isEchoOk) return; // Echo already ok, test just in case ...
223 
224  if (echo.isEmpty()) {
225  if (Arn::debugDepend) qDebug() << "echoCheck: Send request monitorName=" << _name << " req=" << _uuid;
226  slot->arnEchoPipe = _uuid; // Dependency request
227  }
228  else {
229  _timerEchoRefresh->start(); // Only when getting echo, to avoid fillup during disconnect
230  }
231  if (echo == _uuid) { // Dependency echo ok
232  slot->isEchoOk = true;
233  disconnect( &slot->arnEchoPipe, SIGNAL(changed(QString)), this, SLOT(echoCheck(QString))); // Avoid more signals
234  if (slot->useStateCheck) { // Go on with state check
235  connect( &slot->arnStateId, SIGNAL(changed()), this, SLOT(stateCheck()));
236  connect( &slot->arnStateName, SIGNAL(changed()), this, SLOT(stateCheck()));
237  stateCheck( slot);
238  }
239  else { // This dependency is ok
240  doDepOk( slot);
241  }
242  }
243 }

246 void ArnDepend::stateCheck( DepSlot* slot)
247 {
248  if (slot == 0) {
249  ArnItem* arnItem = qobject_cast<ArnItem*>( sender());
250  if (arnItem) slot = static_cast<DepSlot*>( arnItem->reference());
251  }
252  if (slot == 0) {
253  ArnM::errorLog( QString(tr("Can't get slot for DependStateCheck monitor=")) + _name,
254  ArnError::Undef);
255  return;
256  }
257  if (slot->isStateOk) return; // State already ok
258 
259  if (!slot->stateName.isEmpty() && (slot->arnStateName.toString() == slot->stateName)) {
260  slot->isStateOk = true;
261  doDepOk( slot);
262  }
263  else if ((slot->stateId > 0) && (slot->arnStateId.toInt() >= slot->stateId)) {
264  slot->isStateOk = true;
265  doDepOk( slot);
266  }
267 }

280 void ArnDepend::deleteSlot( void* slot_)
281 {
282  DepSlot* slot = reinterpret_cast<DepSlot*>( slot_);
283  if (Arn::debugDepend) qDebug() << "deleteSlot monitorName=" << _name;
284  if (!_depTab.removeOne( slot)) {
285  ArnM::errorLog( QString(tr("Can't get slot for delete Depend monitor=")) + _name,
286  ArnError::Undef);
287  return;
288  }
289  delete slot;
290 
291  if (_depTab.isEmpty()) { // All dependencys ok
292  _timerEchoRefresh->stop();
293  emit completed();
294  _started = false; // Ok to start again
295  }
